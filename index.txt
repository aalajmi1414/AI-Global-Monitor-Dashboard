<!DOCTYPE html>
<html lang="en" dir="ltr">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Global AI Governance Dashboard</title>
    <script src="https://unpkg.com/react@18/umd/react.production.min.js"></script>
    <script src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>
    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
    <script src="https://cdn.tailwindcss.com"></script>
    <link rel="stylesheet" href="https://cdnjs.cloudflare.com/ajax/libs/font-awesome/6.4.0/css/all.min.css">
    <style>
        @import url('https://fonts.googleapis.com/css2?family=Inter:wght@400;500;600;700&family=Tajawal:wght@400;500;700&display=swap');
        .font-en { font-family: 'Inter', sans-serif; }
        .font-ar { font-family: 'Tajawal', sans-serif; }
        .rtl { direction: rtl; }
        .ltr { direction: ltr; }
    </style>
</head>
<body class="bg-gray-50 text-gray-900">
    <div id="root"></div>

    <script type="text/babel">
        const { useState, useMemo, useEffect } = React;

        const dict = {
            en: {
                title: "Global AI Governance Hub",
                subtitle: "Centralized monitoring for AI legislation and policy",
                toggleLang: "عربي",
                lastUpdated: "Last Checked:",
                filters: "Filters",
                search: "Search...",
                layer1: "Layer 1: Enacted & Published",
                layer2: "Layer 2: Pipeline & In-Progress",
                mapView: "Global Status",
                feedView: "Regulatory Feed",
                statusEnacted: "Enacted/In Force",
                statusPipeline: "Pipeline/Draft",
                sourceOfficial: "Official Gov Source",
                sourceNews: "News Report",
                recentUpdate: "Recent Update",
                loading: "Loading live data..."
            },
            ar: {
                title: "مركز حوكمة الذكاء الاصطناعي العالمي",
                subtitle: "مراقبة مركزية لتشريعات وسياسات الذكاء الاصطناعي",
                toggleLang: "English",
                lastUpdated: "آخر فحص:",
                filters: "التصفيات",
                search: "بحث...",
                layer1: "المستوى 1: معتمد ومنشور",
                layer2: "المستوى 2: قيد التطوير والإجراء",
                mapView: "الحالة العالمية",
                feedView: "سجل التشريعات",
                statusEnacted: "معتمد/ساري",
                statusPipeline: "مسودة/مقترح",
                sourceOfficial: "مصدر حكومي رسمي",
                sourceNews: "تقرير إخباري",
                recentUpdate: "تحديث حديث",
                loading: "جاري تحميل البيانات..."
            }
        };

        const Dashboard = () => {
            const [lang, setLang] = useState('en');
            const [search, setSearch] = useState('');
            const [layerFilter, setLayerFilter] = useState('all');
            const [db, setDb] = useState([]);
            const [loading, setLoading] = useState(true);
            const t = dict[lang];

            useEffect(() => {
                // To prevent caching issues on GitHub Pages, we add a timestamp to the fetch URL
                fetch(`./data.json?t=${new Date().getTime()}`)
                    .then(response => response.json())
                    .then(data => {
                        setDb(data);
                        setLoading(false);
                    })
                    .catch(error => {
                        console.error("Error loading data:", error);
                        setLoading(false);
                    });
            }, []);

            useEffect(() => {
                document.documentElement.dir = lang === 'ar' ? 'rtl' : 'ltr';
                document.documentElement.lang = lang;
            }, [lang]);

            const filteredDb = useMemo(() => {
                return db.filter(item => {
                    const matchesSearch = (item.countryEn + item.titleEn + item.summaryEn + item.countryAr + item.titleAr + item.summaryAr).toLowerCase().includes(search.toLowerCase());
                    const matchesLayer = layerFilter === 'all' || item.layer === parseInt(layerFilter);
                    return matchesSearch && matchesLayer;
                }).sort((a, b) => new Date(b.date) - new Date(a.date));
            }, [search, layerFilter, db]);

            const ItemCard = ({ item }) => (
                <div className={`p-4 mb-4 border rounded-lg bg-white shadow-sm relative ${item.recent ? 'border-blue-500' : 'border-gray-200'}`}>
                    {item.recent && (
                        <span className={`absolute top-0 ${lang === 'ar' ? 'left-0 rounded-tl-lg rounded-br-lg' : 'right-0 rounded-tr-lg rounded-bl-lg'} bg-blue-100 text-blue-800 text-xs px-2 py-1 font-semibold`}>
                            {t.recentUpdate}
                        </span>
                    )}
                    <div className="flex justify-between items-start mb-2 mt-2">
                        <div className="flex items-center gap-2">
                            <span className="text-2xl">{item.flag}</span>
                            <h3 className="font-bold text-lg">{lang === 'ar' ? item.countryAr : item.countryEn}</h3>
                        </div>
                        <span className={`px-2 py-1 text-xs font-semibold rounded-full ${item.layer === 1 ? 'bg-green-100 text-green-800' : 'bg-orange-100 text-orange-800'}`}>
                            {item.layer === 1 ? t.layer1 : t.layer2}
                        </span>
                    </div>
                    <h4 className="font-semibold text-md text-gray-800 mb-1">{lang === 'ar' ? item.titleAr : item.titleEn}</h4>
                    <p className="text-sm text-gray-600 mb-3">{lang === 'ar' ? item.summaryAr : item.summaryEn}</p>
                    <div className="flex justify-between items-center text-xs text-gray-500 border-t pt-3 mt-2">
                        <span><i className="fa-regular fa-calendar mx-1"></i> {item.date}</span>
                        <span><i className="fa-solid fa-scale-balanced mx-1"></i> {lang === 'ar' ? item.statusAr : item.statusEn}</span>
                        
                        {/* THIS IS THE NEW CLICKABLE LINK SECTION */}
                        <a href={item.url} target="_blank" rel="noopener noreferrer" 
                           className={`flex items-center hover:underline ${item.sourceType === 'official' ? 'text-blue-600 font-medium' : 'text-gray-500 italic'}`}
                           title={lang === 'ar' ? 'عرض المستند' : 'View Document'}>
                            <i className={item.sourceType === 'official' ? "fa-solid fa-landmark mx-1" : "fa-regular fa-newspaper mx-1"}></i>
                            {item.source}
                            <i className="fa-solid fa-arrow-up-right-from-square mx-1 text-[10px]"></i>
                        </a>
                        {/* END OF LINK SECTION */}

                    </div>
                </div>
            );

            return (
                <div className={`min-h-screen ${lang === 'ar' ? 'font-ar rtl' : 'font-en ltr'}`}>
                    <header className="bg-slate-900 text-white p-4 shadow-md sticky top-0 z-10">
                        <div className="max-w-7xl mx-auto flex justify-between items-center">
                            <div>
                                <h1 className="text-2xl font-bold">{t.title}</h1>
                                <p className="text-slate-400 text-sm">{t.subtitle}</p>
                            </div>
                            <div className="flex items-center gap-4">
                                <button onClick={() => setLang(lang === 'en' ? 'ar' : 'en')} className="bg-slate-700 hover:bg-slate-600 px-4 py-2 rounded-md font-medium transition-colors">
                                    {t.toggleLang}
                                </button>
                            </div>
                        </div>
                    </header>

                    <main className="max-w-7xl mx-auto p-6 grid grid-cols-1 lg:grid-cols-3 gap-6">
                        <div className="lg:col-span-1 space-y-6">
                            <div className="bg-white p-5 rounded-xl shadow-sm border border-gray-200">
                                <h2 className="font-bold text-lg mb-4">{t.filters}</h2>
                                <input type="text" placeholder={t.search} className="w-full p-2 border border-gray-300 rounded-md mb-4 bg-gray-50" onChange={(e) => setSearch(e.target.value)} />
                                <div className="space-y-2">
                                    <label className="flex items-center gap-2 cursor-pointer"><input type="radio" name="layer" value="all" checked={layerFilter === 'all'} onChange={() => setLayerFilter('all')} /><span>All</span></label>
                                    <label className="flex items-center gap-2 cursor-pointer text-green-700 font-medium"><input type="radio" name="layer" value="1" checked={layerFilter === '1'} onChange={() => setLayerFilter('1')} /><span>{t.layer1}</span></label>
                                    <label className="flex items-center gap-2 cursor-pointer text-orange-700 font-medium"><input type="radio" name="layer" value="2" checked={layerFilter === '2'} onChange={() => setLayerFilter('2')} /><span>{t.layer2}</span></label>
                                </div>
                            </div>
                        </div>

                        <div className="lg:col-span-2">
                            <h2 className="font-bold text-xl mb-4 text-gray-800 border-b pb-2">{t.feedView}</h2>
                            {loading ? (
                                <div className="text-center p-10 text-gray-500"><i className="fa-solid fa-spinner fa-spin text-2xl mb-2"></i><p>{t.loading}</p></div>
                            ) : (
                                <div className="space-y-4">
                                    {filteredDb.map(item => <ItemCard key={item.id} item={item} />)}
                                    {filteredDb.length === 0 && <div className="text-center text-gray-500 p-8 border border-dashed rounded-lg">No records found.</div>}
                                </div>
                            )}
                        </div>
                    </main>
                </div>
            );
        };

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<Dashboard />);
    </script>
</body>
</html>
